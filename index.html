const URL = "https://teachablemachine.withgoogle.com/models/WBQm5pH1S/"; let model, webcam, labelContainer, maxPredictions; let isRunning = false; let loopTimeout; // เพิ่มตัวแปรสำหรับเก็บค่าการตั้งเวลา async function init() { if (isRunning) return; isRunning = true; const startBtn = document.getElementById("start-btn"); startBtn.disabled = true; startBtn.innerText = "กำลังโหลดโมเดล..."; try { const modelURL = URL + "model.json"; const metadataURL = URL + "metadata.json"; model = await tmImage.load(modelURL, metadataURL); maxPredictions = model.getTotalClasses(); const flip = true; webcam = new tmImage.Webcam(200, 200, flip); await webcam.setup(); await webcam.play(); const webcamContainer = document.getElementById("webcam-container"); webcamContainer.innerHTML = ""; webcamContainer.appendChild(webcam.canvas); labelContainer = document.getElementById("label-container"); labelContainer.innerHTML = ""; for (let i = 0; i < maxPredictions; i++) { const row = document.createElement("div"); row.className = "prediction-item"; row.innerHTML = `...0.0%`; labelContainer.appendChild(row); } startBtn.innerText = "กำลังทำงาน (ทุกๆ 3 วิ)..."; // เรียกฟังก์ชันลูปครั้งแรก loop(); } catch (error) { console.error("เกิดข้อผิดพลาด:", error); alert("ไม่สามารถเปิดกล้องได้"); startBtn.disabled = false; startBtn.innerText = "ลองใหม่อีกครั้ง"; isRunning = false; } } // เปลี่ยนมาใช้ async loop ร่วมกับ setTimeout เพื่อหน่วงเวลา 3 วินาที async function loop() { if (!isRunning) return; try { webcam.update(); // อัปเดตภาพจากกล้อง ณ วินาทีนั้น await predict(); // ทำนายผลทันที } catch (err) { console.error("ลูปทำงานผิดพลาด:", err); } // เมื่อทำนายเสร็จแล้ว ให้รออีก 3000 มิลลิวินาที (3 วินาที) ก่อนจะเรียกตัวเองใหม่อีกครั้ง loopTimeout = setTimeout(loop, 3000); } async function predict() { const prediction = await model.predict(webcam.canvas); for (let i = 0; i < maxPredictions; i++) { const item = labelContainer.childNodes[i]; const classNameText = prediction[i].className; const probabilityText = (prediction[i].probability * 100).toFixed(1) + "%"; item.querySelector(".class-name").textContent = classNameText; item.querySelector(".probability").textContent = probabilityText; } }
